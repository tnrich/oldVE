                }

                loc = "complement(" + loc + ")"; 
            }

            for (var i=0; i < featureQualifier.length; i++) {
                qual += featureQualifier[i].toString();
                if (i<featureQualifier.length - 1) { 
                    qual += "\n";
                }
            }

            line = line + loc + "\n" + qual;
            return line;
